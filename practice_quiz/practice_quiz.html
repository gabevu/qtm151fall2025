# Write your answer here:
import numpy as np
import matplotlib.pyplot as plt
x_values = np.linspace(0, 2*np.pi, 100)
vec_y1 = np.sin(x_values)
vec_y2 = np.cos(x_values)
list_vectors = [vec_y1, vec_y2]

for vector in list_vectors:
    plt.plot(x_values, vector)

plt.title("Graph 1: Sine and Cosine Waves between 0 and 2pi")
plt.xlabel("Value")
plt.ylabel("Frequency")
plt.show()

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

#create a csv file of the existing data
csv_files = ["practice_quiz_data/korea_rep_co2_emissions.csv", "practice_quiz_data/united_states_co2_emissions.csv"]

#set up an empty list 
dataframes = []

#create the for loop to read each csv file and append it to the empty list 
for file in csv_files: 
    df = pd.read_csv(file) #reads the csv file
    mean_value = df["total_emissions"].mean() #calculates the mean of the total_emissions column
    dataframes.append(mean_value) #appends the mean value to the empty list
